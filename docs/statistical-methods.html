2.4 Statistical Methods

2.4.1 Linear Regression

Summarise and perform linear regression on multiple groups

mtcars %>%  group_by(as.factor(gear)) %>%
  summarise(mean = mean(qsec), sd = sd(qsec))
## # A tibble: 3 x 3
##   `as.factor(gear)`  mean    sd
##   <fct>             <dbl> <dbl>
## 1 3                  17.7  1.35
## 2 4                  19.0  1.61
## 3 5                  15.6  1.13
#What if you wanted to run the same linear regression model by the different
#group levels?  Instead of running #summary(lm(y~x)) for the number of levels 
#you have, you can use the R package “broom” along with dplyr.

# Run the same regression model for gears ##
kable(mtcars%>% group_by(gear) %>%
  do(fitgear = glance(lm(hp~qsec, data = .))) %>% 
  unnest(fitgear)) %>% kable_minimal()

2.4.2 Summary Tables

Summarise by group

kable(mtcars %>% group_by(cyl) %>% summarise(Average=mean(hp), StDev=sd(hp))) %>% 
  kable_styling(full_width = FALSE) %>% kable_minimal()

Summarise multiple functions across several variables

# make sure brackets are correct

df.sum <- mtcars %>%  select(mpg,cyl,hp) %>%
  summarise(across(everything(),list(mean=mean,sd=sd)))
kable(df.sum) %>% kable_styling(full_width = FALSE) %>% kable_minimal()
mpg_mean mpg_sd cyl_mean cyl_sd hp_mean hp_sd
20.09062 6.026948 6.1875 1.785922 146.6875 68.56287 
df.longer <- df.sum%>% pivot_longer(col=everything(),  
names_to = c("Attribute",".value"),
  names_sep = "_")
kable(df.longer) %>% kable_styling(full_width = FALSE) %>% kable_minimal()
